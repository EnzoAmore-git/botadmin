    function switchTab(tabName) {
        document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
        document.querySelectorAll('.tab').forEach(el => el.classList.remove('active'));
        document.getElementById(tabName).classList.add('active');
        event.target.classList.add('active');
    }

    function toggleDataset(datasetName) {
        activeDatasets[datasetName] = !activeDatasets[datasetName];

        document.querySelectorAll('.kpi-card').forEach(card => {
            if (card.dataset.dataset === datasetName) {
                card.classList.toggle('disabled', !activeDatasets[datasetName]);
                card.classList.toggle('active', activeDatasets[datasetName]);
            }
        });

        updateChartLegend();
        updateDailyChart();
    }

    function updateChartLegend() {
        const legend = document.getElementById('chartLegend');
        legend.innerHTML = '';

        const legendItems = [
            { id: 'total', label: 'Всего звонков', color: '#667eea' },
            { id: 'withDialog', label: 'С диалогом', color: '#4ecdc4' },
            { id: 'withoutDialog', label: 'Без диалога', color: '#ff6b6b' },
            { id: 'clientHangups', label: 'Клиент сбросил', color: '#ffe66d' },
            { id: 'botHangups', label: 'Бот завершил', color: '#a8e6cf' }
        ];

        legendItems.forEach(item => {
            const div = document.createElement('div');
            div.className = `legend-item ${activeDatasets[item.id] ? '' : 'disabled'}`;
            div.onclick = () => toggleDataset(item.id);
            div.innerHTML = `
                    <div class="legend-color" style="background: ${item.color}"></div> 
                    <span>${item.label}</span> 
                `;
            legend.appendChild(div);
        });
    }

    function scrollChart(direction) {
        const wrapper = document.getElementById('chartWrapper');
        const scrollAmount = 200;
        if (direction === 'left') {
            chartScrollPosition = Math.max(0, chartScrollPosition - scrollAmount);
        } else {
            const maxScroll = wrapper.scrollWidth - wrapper.clientWidth;
            chartScrollPosition = Math.min(maxScroll, chartScrollPosition + scrollAmount);
        }

            card.dataset.dataset = kpi.dataset || '';
            card.style.setProperty('--card-color', kpi.color);
            if (kpi.dataset) {
                card.onclick = () => toggleDataset(kpi.dataset);
            }

            let badge = '';
            const value = typeof kpi.value === 'number' ? kpi.value : parseFloat(kpi.value);
            if (kpi.id === 'conversion') {
                badge = '<span class="metric-badge bad">Низкая</span>';
            } else if (kpi.id === 'withoutDialog') {
                badge = '<span class="metric-badge bad">Критично</span>';
            } else if (kpi.id === 'clientHangups') {
                badge = '<span class="metric-badge warning">Высокий</span>';
            }

            // Calculate delta with proper sign and color
            let deltaHtml = '';
            if (kpi.prevValue !== undefined) {
                const prev = typeof kpi.prevValue === 'number' ? kpi.prevValue : parseFloat(kpi.prevValue);
                const current = typeof kpi.value === 'number' ? kpi.value : parseFloat(kpi.value);
                const delta = current - prev;
                const deltaPercent = ((delta / prev) * 100).toFixed(1);

                // Determine if this is good or bad based on inverseLogic
                const isGoodChange = kpi.inverseLogic ? (delta <= 0) : (delta >= 0);
                const deltaClass = isGoodChange ? 'positive' : 'negative';
                const deltaSign = delta >= 0 ? '+' : '';
                const arrow = delta >= 0 ? '↗' : '';

                deltaHtml = `<div class="kpi-delta ${deltaClass}">${arrow} ${deltaSign}${deltaPercent}% к прошлой неделе</div>`;
            }

                    <div class="kpi-label">${kpi.label}</div> 
                    <div class="kpi-value">${typeof kpi.value === 'number' ? kpi.value.toLocaleString() : kpi.value}${badge}</div>
